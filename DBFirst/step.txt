-- Tạo db sql đầu tiên

B1: tạo project mvc = cli 
dotnet new mvc -n DBFirst
B2 cài đặt các gói 
dotnet add package Microsoft.EntityFrameworkCore.SqlServer

dotnet add package Microsoft.EntityFrameworkCore.Design

dotnet add package Microsoft.EntityFrameworkCore.Tools

b3: thực hiện DBFirst
dotnet ef dbcontext scaffold "Server=.;Database=NameDB;Trusted_Connection=True;TrustServerCertificate=True" Microsoft.EntityFrameworkCore.SqlServer -o Models
